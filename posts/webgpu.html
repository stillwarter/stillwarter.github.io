async function draw(
  device: GPUDevice,
  context: GPUCanvasContext,
  pipeline: GPURenderPipeline
) {

  const commandEncoder = device.createCommandEncoder();

        storeOp: "store",
      },
    ],
  };
  // иЃЊзљЃrenderpass

  // дљњзФ®еСљдї§еѓєpipelineињЫи°МеК†иљљ
